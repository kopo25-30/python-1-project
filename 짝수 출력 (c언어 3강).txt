#include <stdio.h>

void main() {
    
    int i;
    int j = 1;
    
    printf("숫자를 입력하세요. : "); 
    scanf("%d", &i);
    
    if (i <= 0) {
        printf("1부터 입력해주세요.");
    } else if (i % 2 != 0) {
        printf("짝수만 입력 할 수 있습니다.");
    } else {
    while (j < i) {
        if (j == 6) {
            break;
        }
        printf("%d ", j * 2);
        j++;
    }
    }
}
