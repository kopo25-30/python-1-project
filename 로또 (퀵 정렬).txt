#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <unistd.h>

void quick(int arr[], int left, int right) {
    if (left >= right) return;

    int pivot = arr[left];
    int i = left + 1;
    int j = right;
    int temp;

    while (i <= j) {
        while (i <= right && arr[i] <= pivot) i++;
        while (j >= left + 1 && arr[j] >= pivot) j--;
        if (i < j) {
            temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }

    // 피벗과 j 위치 교환
    temp = arr[left];
    arr[left] = arr[j];
    arr[j] = temp;

    quick(arr, left, j - 1);
    quick(arr, j + 1, right);
}

int main() {
    int num[6];
    srand(time(NULL));

    for (int i = 0; i < 6; i++) {
        num[i] = (rand() % 45) + 1;

        // 중복 체크
        for (int k = 0; k < i; k++) {
            if (num[i] == num[k]) {
                i--;
                break;
            }
        }
    }

    // 퀵 정렬 호출
    quick(num, 0, 5);

    // 결과 출력
    for (int i = 0; i < 6; i++) {
        printf("%d\n", num[i]);
        sleep(1); // 1초 간격 출력
    }

    return 0;
}
