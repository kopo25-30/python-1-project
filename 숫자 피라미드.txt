n = int(input("숫자를 입력해주세요: "))

if n > 8:
    print("8까지만 출력할 수 있습니다.")
elif n < 1:
    print("1 이상의 숫자를 입력해주세요.")
else:
    num = 1       # 출력할 숫자 시작
    count = 1     # 한 줄에 출력할 숫자 개수 (1 → 3 → 5 → ...)

    while num <= n:
        line = ''
        for _ in range(count):
            if num > n:
                break
            line += str(num)
            num += 1

        print(line.center(20))  # 가운데 정렬
        count += 2
